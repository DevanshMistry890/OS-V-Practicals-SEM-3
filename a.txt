a.	To perform mathematical operations.
Script:
echo "Enter 1st Number" 
read num1
echo “Enter 2nd Number”
read num2
sum=`expr $num1 + $num2`
sub=`expr $num1 - $num2`
mul=$(($num1*$num2))
div=$(($num1/$num2))
echo “Addition is $sum”
echo “subtraction is $sub”
echo “multiplication is $mul”
echo “division is $div”
